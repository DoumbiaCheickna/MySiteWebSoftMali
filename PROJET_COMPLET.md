```
│   ├── contact.html           # Contact + formulaire

│   └── coming-soon.html       # Page "Bientôt disponible"

│   └── css/
│       └── style.css          # CSS principal (responsive)
│
├── ⚡ SCRIPTS
│   └── js/
│       └── main.js            # JavaScript interactif
│
├── 🖼️ MÉDIAS
│   ├── images/                # Vos images ici
│   └── assets/                # Favicon et ressources
│
├── 🔧 CONFIGURATION
│   ├── .htaccess              # Config Apache (sécurité, cache)
│   ├── .gitignore             # Fichiers à ignorer par Git
│   ├── robots.txt             # Instructions moteurs de recherche
│   └── sitemap.xml            # Plan du site pour SEO
│
└── 📚 DOCUMENTATION
    ├── README.md                      # Documentation principale
    ├── GUIDE_DEMARRAGE.md            # Démarrage rapide
    ├── GUIDE_DEPLOIEMENT.md          # Guide de mise en ligne
    ├── CHECKLIST_PERSONNALISATION.md # Checklist complète
    └── SEO_CONFIG.md                 # Configuration SEO

```
